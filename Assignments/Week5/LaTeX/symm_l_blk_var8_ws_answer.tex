% Define the operation to be computed

\renewcommand{\operation}{ C := A B + C \mbox{~~~where $ A $ is symmetric and stored in the lower triangular part} }

\renewcommand{\routinename}{\operation}

% Step 1a: Precondition 

\renewcommand{\precondition}{
	C = \widehat{C}
}

% Step 1b: Postcondition 

\renewcommand{\postcondition}{ 
	C = A B + \widehat{C} 
}

% Step 2: Invariant 
% Note: Right-hand side of equalities must be updated appropriately

\renewcommand{\invariant}{
	\left(\begin{array}{c}
		C_T \\ \whline
		C_B 
	\end{array}\right)  = 
	\left(\begin{array}{c}
		{\color{white} A_{TL} B_T + A_{BL}^T B_B +} \widehat C_T \\ \whline 
		{\color{white} A_{BL} B_T +} A_{BR} B_B + \widehat C_B 
	\end{array}\right)
}

% Step 3: Loop-guard 

\renewcommand{\guard}{
	m( A_{BR} ) < m( A )
}

% Step 4: Initialize 

\renewcommand{\partitionings}{
	$
	A \rightarrow
	\left(\begin{array}{c I c} 
	A_{TL} & A_{TR} \\ \whline
	A_{BL} & A_{BR} 
	\end{array}\right) 
	$
	,
	$
	B \rightarrow
	\left(\begin{array}{c}
	B_{T} \\ \whline 
	B_{B}
	\end{array}\right)
	$
	,
	$
	C \rightarrow
	\left(\begin{array}{c}
	C_{T} \\ \whline 
	C_{B}
	\end{array}\right)
	$
}

\renewcommand{\partitionsizes}{
	$ A_{BR} $ is $ 0 \times 0 $,
	$ B_B $ has $ 0 $ rows,
	$ C_B $ has $ 0 $ rows
}

% Step 5a: Repartition the operands 

\renewcommand{\blocksize}{b}

\renewcommand{\repartitionings}{
	$  \left(\begin{array}{c I c}
	A_{TL} & A_{TR} \\ \whline 
	A_{BL} & A_{BR}
	\end{array}\right) 
	\rightarrow
	\left(\begin{array}{c c I c}
	A_{00} & A_{01} & A_{02} \\ 
	A_{10} & A_{11} & A_{12} \\  \whline 
	A_{20} & A_{21} & A_{22}
	\end{array}\right) 
	$
	,
	$  \left(\begin{array}{c}
	B_T \\ \whline
	B_B 
	\end{array}\right) 
	\rightarrow
	\left(\begin{array}{c}
	B_0 \\ 
	B_1 \\  \whline 
	B_2
	\end{array}\right)
	$
	,
	$  \left(\begin{array}{c}
	C_T \\ \whline
	C_B 
	\end{array}\right) 
	\rightarrow
	\left(\begin{array}{c}
	C_0 \\  
	C_1 \\  \whline
	C_2
	\end{array}\right)
	$
}

\renewcommand{\repartitionsizes}{
	$ A_{11} $ is $ b \times b $,
	$ B_1 $ has $ b $ rows,
	$ C_1 $ has $ b $ rows}

% Step 5b: Move the double lines 

\renewcommand{\moveboundaries}{
	$  \left(\begin{array}{c I c}
	A_{TL} & A_{TR} \\ \whline 
	A_{BL} & A_{BR}
	\end{array}\right) 
	\leftarrow
	\left(\begin{array}{c I c c}
	A_{00} & A_{01} & A_{02} \\  \whline 
	A_{10} & A_{11} & A_{12} \\ 
	A_{20} & A_{21} & A_{22}
	\end{array}\right) 
	$
	,
	$  \left(\begin{array}{c}
	B_T \\ \whline
	B_B 
	\end{array}\right) 
	\leftarrow
	\left(\begin{array}{c}
	B_0 \\  \whline 
	B_1 \\ 
	B_2
	\end{array}\right) 
	$
	,
	$  \left(\begin{array}{c}
	C_T \\ \whline
	C_B 
	\end{array}\right) 
	\leftarrow
	\left(\begin{array}{c}
	C_0 \\  \whline
	C_1 \\  
	C_2
	\end{array}\right) 
	$
}

\renewcommand{\beforeupdate}{$
	\left(\begin{array}{c}
	C_0 \\ 
	C_1 \\
	C_2 
	\end{array}\right)  = 
	\left(\begin{array}{r}
	{\color{white} A_{00} B_0 + A_{10}^T B_1 + A_{20}^T B_2 +} \widehat C_0 \\ 
	{\color{white} A_{10} B_0 + A_{11} B_1 + A_{21}^T B_2 +} \widehat C_1 \\ 
	{\color{white} A_{20} B_0 + A_{21} B_1 +} A_{22} B_2 + \widehat C_2 
	\end{array}\right)
	$}


% Step 7: State after update

\renewcommand{\afterupdate}{$

	C_0 \\ 
	C_1 \\
	C_2 
	\end{array}\right)  = 
	\left(\begin{array}{r}
	{\color{white} A_{00} B_0 + A_{10}^T B_1 + A_{20}^T B_2 +} \widehat C_0 \\ 
	{\color{white} A_{10} B_0 +} A_{11} B_1 + A_{21}^T B_2 + \widehat C_1 \\ 
	{\color{white} A_{20} B_0 +} A_{21} B_1 + A_{22} B_2 + \widehat C_2 
	\end{array}\right)
	$}
